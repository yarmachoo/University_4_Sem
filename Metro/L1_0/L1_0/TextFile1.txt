use strict;
use warnings;

my $eps = 0.0001;
my ($y, $x, $n, $vs);

print "Enter x: ";
$x = <STDIN>;
chomp $x;

$y = $x; # Начальные установки
$n = 2;
$vs = $x;

while (abs($vs) >= $eps) { # Выход из цикла по выполнению условия
    $vs = -$vs * $x * $x / (2 * $n - 1) / (2 * $n - 2); # Формирование слагаемого
    $n++;
    $y += $vs;
}

print "$x $y $eps\n";
